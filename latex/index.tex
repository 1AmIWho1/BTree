For creating \mbox{\hyperlink{classBTree}{B\+Tree}} you should use standart constructor. This \mbox{\hyperlink{classBTree}{B\+Tree}} implementation has minimum t keys in every \mbox{\hyperlink{classNode}{Node}}, and maximum 2t. t is order of \mbox{\hyperlink{classBTree}{B\+Tree}}.

Here is a simple example of inserting keys in \mbox{\hyperlink{classBTree}{B\+Tree}}\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{\mbox{\hyperlink{classBTree}{BTree}} b = \mbox{\hyperlink{classBTree}{BTree}}(2);}
\DoxyCodeLine{b.\mbox{\hyperlink{classBTree_aa17aba464822992d9ae5b316b1875eae}{Insert}}(3);}
\DoxyCodeLine{b.\mbox{\hyperlink{classBTree_aa17aba464822992d9ae5b316b1875eae}{Insert}}(22);}
\DoxyCodeLine{b.\mbox{\hyperlink{classBTree_aa17aba464822992d9ae5b316b1875eae}{Insert}}(10);}
\DoxyCodeLine{b.\mbox{\hyperlink{classBTree_aa17aba464822992d9ae5b316b1875eae}{Insert}}(2);}
\DoxyCodeLine{b.\mbox{\hyperlink{classBTree_aa17aba464822992d9ae5b316b1875eae}{Insert}}(17);}
\DoxyCodeLine{b.\mbox{\hyperlink{classBTree_aa17aba464822992d9ae5b316b1875eae}{Insert}}(5);}
\end{DoxyCode}
